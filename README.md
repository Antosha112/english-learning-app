##все значения cursor с примерами

.clickable {
    cursor: pointer;    /* 👆 Рука с пальцем - стандарт для ссылок */
}


.disabled {
    cursor: not-allowed;   /* 🚫 Запрещено (круг с чертой) */
}


.input-field {
    cursor: text;          /* I-образный курсор для текста */
}


.draggable {
    cursor: grab;          /* ✋ Рука (захват) */
}


.dragging {
    cursor: grabbing;      /* ✊ Сжатая рука */
}


.action {
    cursor: default;       /* ↖ Обычная стрелка (по умолчанию) */
}


.dropdown {
    cursor: context-menu;  /* 📋 Стрелка с меню */
}
